SET(PROJECT_EXAMPLE_NAME buildModel)

PROJECT(${PROJECT_EXAMPLE_NAME}_app)

FILE(GLOB PROJECT_SOURCES "*.cpp")

FILE(GLOB PROJECT_INCLUDES "*.h")

ADD_EXECUTABLE(${PROJECT_EXAMPLE_NAME}_app ${PROJECT_SOURCES} ${PROJECT_INCLUDES})

TARGET_LINK_LIBRARIES(${PROJECT_EXAMPLE_NAME}_app ifc2x3)

if(NOT WIN32)
  if(NOT APPLE)
    TARGET_LINK_LIBRARIES(${PROJECT_EXAMPLE_NAME}_app uuid)
  endif()
else()
    TARGET_LINK_LIBRARIES(${PROJECT_EXAMPLE_NAME}_app rpcrt4)
endif()

SET_TARGET_PROPERTIES(${PROJECT_EXAMPLE_NAME}_app PROPERTIES PROJECT_LABEL "${PROJECT_EXAMPLE_NAME}")
SET_TARGET_PROPERTIES(${PROJECT_EXAMPLE_NAME}_app PROPERTIES FOLDER "${CURRENT_FOLDER_LIST}")
